cmake_minimum_required (VERSION 2.6)
project (newuoa)
set (CMAKE_C_FLAGS "-Wall -lm -O3 -g")
include_directories (include)
add_library (newuoa_static STATIC src/newuoa.c include/newuoa.h)
set_target_properties (newuoa_static PROPERTIES OUTPUT_NAME newuoa)
add_library (newuoa_shared SHARED src/newuoa.c include/newuoa.h)
set_target_properties (newuoa_shared PROPERTIES OUTPUT_NAME newuoa)
add_executable (example src/example.c include/newuoa.h)
target_link_libraries (example newuoa_static)
